Source code for fastf1.livetiming

"""
:mod:`fastf1.livetiming` - Live Timing Client
=====================================================

This module can be used to save live timing data during a session.

.. warning:: Receiving and saving the SignalR live timing data is
    experimental. This functionality is largely untested due to a distinct
    lack of testing opportunities.

.. note:: While data can be saved, it is not yet possible to actually use
    the saved data. An update which allows using the data will likely be
    available after the 2021 pre-season testing.


There are two ways to interact with this module.

    - From a python script by creating an instance of
      :class:`SignalRClient`

    - From the command line by calling ``python -m fastf1.livetiming``
"""


from datetime import timedelta
import json

from fastf1.utils import to_datetime



[docs]class LivetimingDataResponse:
    """Livetiming data object for use with the api calls.

    This object is created from data that was recorded using
    :class:`fastf1.livetiming.SignalRClient`. It can be passed to
    api call functions using the ``response`` keyword.

    See :mod:`fastf1.livetiming` for a usage example.

    Args:
        fname (str) : file name (opt. with path);
            file from which to read the live timing data from
        reference_zero (datetime.datetime) : the timestamp which should
            be used as the zero reference for the data; defaults to the
            timestamp of the first message
    """
    def __init__(self, fname, reference_zero=None):
        self.data_by_category = dict()
        self.reference_zero = reference_zero
        self.errorcount = 0

        with open(fname, 'r') as fobj:
            data = fobj.readlines()

        prev_td = timedelta(0)

        for elem in data:
            # fix F1's not json compliant data
            elem = elem.replace("'", '"')\
                .replace('True', 'true')\
                .replace('False', 'false')
            # load the three parts of each data element
            try:
                cat, msg, dt = json.loads(elem)
            except json.JSONDecodeError as e:
                self.errorcount += 1
                continue
            # convert string to datetime; need to strip trailing 'Z',
            # sometimes zero-pad right and sometimes cut to us precision
            dt = to_datetime(dt)
            if self.reference_zero is None:
                self.reference_zero = dt
                td = timedelta(seconds=0)
            else:
                td = dt - self.reference_zero

            # skip wrong timestamps; TODO: maybe fix?
            if td < prev_td:
                continue
            prev_td = td

            if cat not in self.data_by_category:
                self.data_by_category[cat] = [(td, msg), ]
            else:
                self.data_by_category[cat].append((td, msg))
